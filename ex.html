#count mutation
dna_1=""
dna_2=""
d_1=list(dna_1)
d_2=list(dna_2)
n=0
for i in range(len(d_1)):
    if d_1[i]==d_2[i]:
        n+=1
print(len(d_1)-n)

def f(k,m,n):
    s=k+m+n
    d2=(n/s)*((n-1)/(s-1))
    d3=(m/s)*((m-1)/(s-1))*0.25
    d4=(n/s)*(m/(s-1))
    return 1-d2-d3-d4
print(f(30,17,23))

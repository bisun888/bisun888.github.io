def fib(n,m):
    f1,f2 = 1,1
    d=list()
    for i in range(m-2):
        d.append(0)
    d.append(1)
    d.append(0)
    born=0
    for i in range(3,n+1):
        born=f2-born
        f1=f2
        d.append(born)
        f2=f1+born-d[i-3]
    return f2
print(fib(93,17))
